## C Header & Go Wrapper Export Example

This example demonstrates how to use `clue/wrapper` to generate a C header
and Go wrapper from a Nim library at compile time.

### Usage

The `mathlib.nim` module defines a 3D vector math library with `{.exportc.}`
markers. When compiled with `--app:lib -d:clueBuild`, wrappers are generated
at `wrappers/c/mathlib.h` and `wrappers/go/mathlib/mathlib.go`.

### Build

```
nim c mathlib.nim
```

This produces:
- `build/libmathlib.so` — shared library
- `wrappers/c/mathlib.h` — generated C header
- `wrappers/go/mathlib/mathlib.go` — generated Go wrapper
- `wrappers/rust/mathlib/` — generated Rust crate
- `wrappers/crystal/mathlib/` — generated Crystal shard

### Compile and run C test

```
cc -o build/test_mathlib test_mathlib.c -L build -lmathlib -lm
DYLD_LIBRARY_PATH=build ./build/test_mathlib
```

### Compile and run C++ test

Since the generated header includes `extern "C"` guards, it works with C++
compilers without any changes:

```
c++ -o build/test_mathlib test_mathlib.c -L build -lmathlib -lm
DYLD_LIBRARY_PATH=build ./build/test_mathlib
```

### Compile and run Go test

The Go wrapper uses cgo to call the Nim shared library.
First build the Nim library (if not already done):

```
nim c mathlib.nim
```

Then run the Go test:

```
cd wrappers/go/mathlib && go mod init mathlib 2>/dev/null; cd ../../..
CGO_ENABLED=1 CGO_LDFLAGS="-L build -lmathlib" go run test_mathlib.go
```

Or build a Go binary for easier reuse:

```
cd wrappers/go/mathlib && go mod init mathlib 2>/dev/null; cd ../../..
CGO_ENABLED=1 CGO_LDFLAGS="-L build -lmathlib" go build -o build/test_mathlib test_mathlib.go
DYLD_LIBRARY_PATH=build ./build/test_mathlib
```

### Generated C header

```c
/* Auto-generated by Clue. Do not edit. */

#ifndef MATHLIB_H
#define MATHLIB_H

#ifdef __cplusplus
extern "C" {
#endif

typedef struct {
  double x;
  double y;
  double z;
} Vector;

typedef enum {
  MathOp_opAdd,
  MathOp_opSub,
  MathOp_opMul,
  MathOp_opDiv
} MathOp;

/**
 * Add two vectors, store result in `res`
 */
MATHLIB_API void vec_add(Vector* a, Vector* b, Vector* res);

/**
 * Compute the length (magnitude) of a vector
 */
MATHLIB_API double vec_length(Vector* v);

/**
 * Scale a vector in-place by a factor
 */
MATHLIB_API void vec_scale(Vector* v, double factor);

/**
 * Compute the dot product of two vectors
 */
MATHLIB_API double vec_dot(Vector* a, Vector* b);

#ifdef __cplusplus
}
#endif

#endif /* MATHLIB_H */
```

### Generated Go wrapper

```go
package mathlib

/*
#include "../../c/mathlib.h"
*/
import "C"
import "unsafe"

type Vector struct {
	X float64
	Y float64
	Z float64
}

const (
	OpAdd = C.MathOp_opAdd
	OpSub = C.MathOp_opSub
	OpMul = C.MathOp_opMul
	OpDiv = C.MathOp_opDiv
)

// Add two vectors, store result in `res`
func Vec_add(a *Vector, b *Vector, res *Vector) {
	C.vec_add((*C.Vector)(unsafe.Pointer(a)), (*C.Vector)(unsafe.Pointer(b)), (*C.Vector)(unsafe.Pointer(res)))
}

// Compute the length (magnitude) of a vector
func Vec_length(v *Vector) float64 {
	return float64(C.vec_length((*C.Vector)(unsafe.Pointer(v))))
}

// Scale a vector in-place by a factor
func Vec_scale(v *Vector, factor float64) {
	C.vec_scale((*C.Vector)(unsafe.Pointer(v)), C.double(factor))
}

// Compute the dot product of two vectors
func Vec_dot(a *Vector, b *Vector) float64 {
	return float64(C.vec_dot((*C.Vector)(unsafe.Pointer(a)), (*C.Vector)(unsafe.Pointer(b))))
}
```

### Generated Crystal bindings

```crystal
# Auto-generated by Clue. Do not edit.

@[Link(ldflags: "-L #{__DIR__}/../../build -lmathlib")]
lib LibMathlib
  @[Extern]
  struct Vector
    x : Float64
    y : Float64
    z : Float64
  end

  enum MathOp
    OpAdd
    OpSub
    OpMul
    OpDiv
  end

  # Add two vectors, store result in `res`
  fun vec_add(a : Vector*, b : Vector*, res : Vector*) : Void

  # Compute the length (magnitude) of a vector
  fun vec_length(v : Vector*) : Float64

  # Scale a vector in-place by a factor
  fun vec_scale(v : Vector*, factor : Float64) : Void

  # Compute the dot product of two vectors
  fun vec_dot(a : Vector*, b : Vector*) : Float64
end
```

### Compile and run Rust test

The Rust bindings use `extern "C"` declarations via the generated crate.

```
cd wrappers/rust/mathlib
cargo run --example test_mathlib
```

### Compile and run Crystal test

The Crystal bindings use `@[Link]` with a `lib` block to declare the FFI.

```
crystal run test_mathlib.cr
```

### Using from C

```c
#include "wrappers/c/mathlib.h"
#include <stdio.h>

int main() {
    Vector a = {1.0, 2.0, 3.0};
    Vector b = {4.0, 5.0, 6.0};
    Vector r;
    vec_add(&a, &b, &r);
    printf("dot = %f\n", vec_dot(&a, &b));
    return 0;
}
```

Compile with:

```
cc -o test test.c -L build -lmathlib
```

### Using from Go

```go
package main

import (
	"fmt"
	"mathlib"
)

func main() {
	a := mathlib.Vector{X: 1.0, Y: 2.0, Z: 3.0}
	b := mathlib.Vector{X: 4.0, Y: 5.0, Z: 6.0}
	var r mathlib.Vector

	mathlib.Vec_add(&a, &b, &r)
	fmt.Printf("dot = %f\n", mathlib.Vec_dot(&a, &b))
}
```

### Using from Rust

```rust
use mathlib::*;

fn main() {
    let a = Vector { x: 1.0, y: 2.0, z: 3.0 };
    let b = Vector { x: 4.0, y: 5.0, z: 6.0 };
    let mut r = unsafe { std::mem::zeroed() };

    unsafe {
        vec_add(&a, &b, &mut r);
        println!("a + b = ({:.1}, {:.1}, {:.1})", r.x, r.y, r.z);
        println!("dot = {:.1}", vec_dot(&a, &b));
    }
}
```
